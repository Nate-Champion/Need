18) Check perfect number
<!-- C++ Program: Check whether a number is perfect -->
#include <iostream>
using namespace std;

int main() {
    int n;
    cout << "Enter number: "; cin >> n;
    int sum = 0;
    for (int i = 1; i <= n/2; ++i)
        if (n % i == 0) sum += i;
    if (sum == n) cout << n << " is a perfect number\n";
    else cout << n << " is not perfect\n";
    return 0;
}
19) Perfect numbers in a range
<!-- C++ Program: Perfect numbers between a and b -->
#include <iostream>
using namespace std;

bool isPerfect(int n) {
    int sum = 0;
    for (int i = 1; i <= n/2; ++i) if (n % i == 0) sum += i;
    return sum == n;
}

int main() {
    int a, b;
    cout << "Enter range (a b): "; cin >> a >> b;
    for (int i = max(1,a); i <= b; ++i)
        if (isPerfect(i)) cout << i << "\n";
    return 0;
}
20) GCD (Euclidean)
<!-- C++ Program: GCD using Euclidean algorithm -->
#include <iostream>
using namespace std;

int gcd(int a, int b) {
    while (b) { int t = a % b; a = b; b = t; }
    return a;
}

int main() {
    int a, b;
    cout << "Enter two numbers: "; cin >> a >> b;
    cout << "GCD = " << gcd(abs(a), abs(b)) << "\n";
    return 0;
}

25) Sum series 1 - x^2/2! + x^4/4! - ... (nth term)
<!-- C++ Program: Alternating even-power series using factorials -->
#include <iostream>
using namespace std;

long long factorial(int m) {
    long long f = 1;
    for (int i = 2; i <= m; ++i) f *= i;
    return f;
}

int main() {
    double x; int n;
    cout << "Enter x and number of terms n: "; cin >> x >> n;
    double sum = 0.0;
    int sign = 1;
    for (int k = 0; k < n; ++k) {
        int p = 2*k; // power 0,2,4,...
        double term = pow(x, p) / factorial(p);
        sum += sign * term;
        sign = -sign;
    }
    cout << "Result = " << sum << "\n";
    return 0;
}
26) Input positives until -1 — count, max, min, avg
<!-- C++ Program: Read positives until -1 -->
#include <iostream>
#include <climits>
using namespace std;

int main() {
    long long x;
    int count = 0;
    long long sum = 0;
    long long mx = LLONG_MIN, mn = LLONG_MAX;
    cout << "Enter positive integers (-1 to stop):\n";
    while (cin >> x && x != -1) {
        if (x > 0) {
            count++;
            sum += x;
            if (x > mx) mx = x;
            if (x < mn) mn = x;
        } else {
            cout << "Ignoring non-positive input\n";
        }
    }
    if (count == 0) cout << "No positive numbers entered\n";
    else {
        cout << "Count=" << count << " Max=" << mx << " Min=" << mn
             << " Average=" << (double)sum / count << "\n";
    }
    return 0;
}

34) Sum of all primes between 1 and n
<!-- C++ Program: Sum of primes up to n -->
#include <iostream>
using namespace std;

bool isPrime(int x){
    if (x < 2) return false;
    for (int i = 2; i*i <= x; ++i) if (x % i == 0) return false;
    return true;
}

int main() {
    int n; cout << "Enter n: "; cin >> n;
    long long sum = 0;
    for (int i = 2; i <= n; ++i) if (isPrime(i)) sum += i;
    cout << "Sum of primes = " << sum << "\n";
    return 0;
}

36–39) Three-digit lottery
<!-- C++ Program: Three-digit lottery rules combined -->
#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    srand((unsigned)time(nullptr));
    int lottery = rand() % 900 + 100; // 100..999
    int guess;
    cout << "Enter your three-digit guess: "; cin >> guess;
    int l1 = lottery/100, l2 = (lottery/10)%10, l3 = lottery%10;
    int g1 = guess/100, g2 = (guess/10)%10, g3 = guess%10;

    cout << "Lottery number: " << lottery << "\n";

    if (guess == lottery) {
        cout << "Exact match: you win $10,000\n";
        return 0;
    }

    // check if all digits match (any order)
    int countMatchAll = 0;
    int usedL[3] = {0,0,0};
    int g[3] = {g1,g2,g3};
    int L[3] = {l1,l2,l3};
    for (int i=0;i<3;i++){
        for(int j=0;j<3;j++){
            if (!usedL[j] && g[i]==L[j]) { usedL[j]=1; countMatchAll++; break; }
        }
    }
    if (countMatchAll == 3) {
        cout << "All digits match (different order): you win $3,000\n";
        return 0;
    }

    // one digit match?
    int oneMatch = 0;
    for (int i=0;i<3;i++) if (g[i]==l1||g[i]==l2||g[i]==l3) oneMatch++;
    if (oneMatch >= 1) {
        cout << "One digit matched: you win $1,000\n";
    } else {
        cout << "No match\n";
    }
    return 0;
}
40) Read integers until 0 — count pos/neg, total & average (exclude zeros)
<!-- C++ Program: Read until 0, compute stats -->
#include <iostream>
using namespace std;

int main() {
    int x;
    int pos = 0, neg = 0;
    long long sum = 0;
    int count = 0;
    cout << "Enter integers (0 to end):\n";
    while (cin >> x && x != 0) {
        if (x > 0) { pos++; sum += x; count++; }
        else if (x < 0) { neg++; sum += x; count++; }
    }
    cout << "Positive: " << pos << " Negative: " << neg << "\n";
    if (count > 0) cout << "Total = " << sum << " Average = " << (double)sum / count << "\n";
    else cout << "No non-zero values entered\n";
    return 0;
}
41) Tuition increase (5% per year)
<!-- C++ Program: Tuition after 10 years and 4-year total starting then -->
#include <iostream>
using namespace std;

int main() {
    double tuition = 10000.0;
    double rate = 0.05;
    // tuition after 10 years
    for (int i = 0; i < 10; ++i) tuition *= (1 + rate);
    cout << "Tuition after 10 years = " << tuition << "\n";
    // total cost for 4 years starting then
    double total = 0;
    double t = tuition;
    for (int j = 0; j < 4; ++j) { total += t; t *= (1 + rate); }
    cout << "Total for years 11-14 = " << total << "\n";
    return 0;
}

    int count = 0;
    for (int i = 100; i <= 200; ++i) {
        bool d5 = (i % 5 == 0), d6 = (i % 6 == 0);

        }
    }
    if (count % 10 != 0) cout << "\n";
    return 0;
}
43) GCD by checking from min(n1,n2) downwards
<!-- C++ Program: GCD by descending check -->
#include <iostream>
using namespace std;

int main() {
    int n1, n2;
    cout << "Enter n1 and n2: "; cin >> n1 >> n2;
    int d = min(abs(n1), abs(n2));
    for (; d >= 1; --d) {
        if (n1 % d == 0 && n2 % d == 0) {
            cout << "GCD = " << d << "\n";
            break;
        }
    }
    return 0;
}
